1. Basic Data Retrieval & Metadata
  	SELECT * FROM products;
	DESC products;
  	SELECT * FROM products LIMIT 4;
  	SELECT * FROM products LIMIT 5;

2. Sorting and Ordering
	SELECT * FROM products ORDER BY stock;
	SELECT * FROM products ORDER BY prod_original_price;
	SELECT * FROM products ORDER BY stock DESC;
	SELECT * FROM products ORDER BY prod_original_price DESC;

3. Aggregate Functions
	SELECT COUNT(*) FROM products;
	SELECT COUNT(*) FROM products WHERE prod_original_price > 1999;
	SELECT COUNT(*) FROM products WHERE prod_original_price > 1000;
	SELECT SUM(stock) FROM products;
	SELECT AVG(stock) FROM products;
	SELECT MAX(stock) FROM products;
	SELECT MIN(stock) FROM products;

4. Range and Set Filtering
	SELECT * FROM products WHERE prod_original_price BETWEEN 200 AND 1000;
	SELECT * FROM products WHERE prod_original_price NOT BETWEEN 200 AND 1000;
	SELECT * FROM products WHERE PROD_NAME BETWEEN 'D' AND 'K';
	SELECT * FROM products WHERE PROD_NAME NOT BETWEEN 'D' AND 'K';
	SELECT * FROM products WHERE PROD_ORIGINAL_PRICE IN (1999, 249);
	SELECT * FROM products WHERE PROD_ORIGINAL_PRICE IN (99, 249, 849.5);
	SELECT * FROM products WHERE CATEGORY IN ('AUDIO', 'TABLETS');
	SELECT * FROM products WHERE CATEGORY NOT IN ('AUDIO', 'TABLETS');

5. Null and Logical Operators
	SELECT * FROM products WHERE PROD_NAME IS NULL;
	SELECT * FROM products WHERE PROD_NAME IS NOT NULL;
	SELECT * FROM products WHERE PROD_NAME = 'IPHONE 15' AND PROD_ORIGINAL_PRICE = 99;
	SELECT * FROM products WHERE PROD_NAME = 'IPHONE 15' OR PROD_ORIGINAL_PRICE = 99;

6. Grouping and Having Clauses
	SELECT COUNT(ID), COUNTRY FROM CUSTOMER GROUP BY COUNTRY;
	SELECT COUNT(ID), CITY FROM CUSTOMER GROUP BY CITY;
	SELECT COUNT(ID), CITY FROM CUSTOMER GROUP BY CITY ORDER BY COUNT(ID);
	SELECT COUNT(ID), CITY FROM CUSTOMER GROUP BY CITY ORDER BY COUNT(ID) DESC;
	SELECT COUNT(ID), CITY FROM CUSTOMER GROUP BY CITY HAVING COUNT(ID) > 1;
	SELECT COUNT(ID), CITY FROM CUSTOMER GROUP BY CITY HAVING COUNT(ID) > 2;

7. Table Joins
	-- Inner Join
	SELECT EMP_NAME, DEPT_NAME FROM EMPLOYEE INNER JOIN DEPARTMENT ON EMPLOYEE.DEPT_ID = DEPARTMENT.DEPT_ID;

	-- Left Joins
	SELECT EMP_NAME, DEPT_NAME FROM EMPLOYEE LEFT JOIN DEPARTMENT ON EMPLOYEE.DEPT_ID = DEPARTMENT.DEPT_ID;
	SELECT EMP_NAME, DEPT_NAME FROM DEPARTMENT LEFT JOIN EMPLOYEE ON EMPLOYEE.DEPT_ID = DEPARTMENT.DEPT_ID;

	-- Right Join
	SELECT EMP_NAME, DEPT_NAME FROM EMPLOYEE RIGHT JOIN DEPARTMENT ON EMPLOYEE.DEPT_ID = DEPARTMENT.DEPT_ID;

	-- Full and Cross Joins
	SELECT EMP_NAME, DEPT_NAME FROM EMPLOYEE FULL JOIN DEPARTMENT;
	SELECT EMP_NAME, DEPT_NAME FROM EMPLOYEE CROSS JOIN DEPARTMENT;
